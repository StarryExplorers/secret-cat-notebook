Cat Notebook body { font-family: 'Comic Sans MS', cursive; background-color: #fff5ee; margin: 0; padding: 20px; } h1, h2 { text-align: center; } #notebook { width: 100%; height: 200px; padding: 10px; font-size: 16px; border: 2px dashed #ccc; background: #fff; margin-bottom: 20px; } .cat-pic { display: inline-block; margin: 10px; cursor: pointer; } .cat-house { display: none; border: 2px solid #aaa; padding: 10px; background-color: #fff; } .cat-actions button { margin: 5px; } .room { display: none; margin-top: 10px; } My Cat Notebook const keywords = ['paws', 'meow', 'whiskers', 'furball', 'purr', 'claws']; const savedCats = JSON.parse(localStorage.getItem('savedCats') || '{}'); const notebook = document.getElementById('notebook'); const catsContainer = document.getElementById('cats'); const houseContainer = document.getElementById('catHouseContainer'); notebook.addEventListener('input', () => { const text = notebook.value.toLowerCase(); keywords.forEach(word => { if (text.includes(word) && !savedCats[word]) { const catName = prompt(`You discovered the word "${word}"! Name your cat:`); if (catName) { savedCats[word] = catName; localStorage.setItem('savedCats', JSON.stringify(savedCats)); createCatIcon(word, catName); } } }); }); function createCatIcon(word, name) { const catDiv = document.createElement('div'); catDiv.className = 'cat-pic'; catDiv.innerHTML = ` ${name} `; catDiv.onclick = () => openCatHouse(word, name); catsContainer.appendChild(catDiv); } function openCatHouse(word, name) { houseContainer.innerHTML = ` ${name}'s House Feed Bath Sleep Dress Play Makeup You gave ${name} some yummy fish! ${name} is now clean and fluffy! ${name} is napping peacefully. ${name} is now wearing a cute outfit! ${name} is chasing a ball of yarn! ${name} has shiny sparkly whiskers now! `; document.querySelectorAll('.cat-house').forEach(div => div.style.display = 'none'); document.getElementById(`house-${word}`).style.display = 'block'; } function toggleRoom(id) { document.querySelectorAll('.room').forEach(div => div.style.display = 'none'); document.getElementById(id).style.display = 'block'; } window.onload = () => { Object.entries(savedCats).forEach(([word, name]) => { createCatIcon(word, name); }); }
